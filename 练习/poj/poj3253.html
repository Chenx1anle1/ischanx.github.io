        printf("%d\n",0);

        while(q.size() >1){
            long long a = q.top();
            q.pop();
            long long b = q.top();
            q.pop();
            const long long temp = a+b;
            sum += temp;
            q.push(temp);
        }
        printf("%lld\n",sum);
    }
    return 0;
