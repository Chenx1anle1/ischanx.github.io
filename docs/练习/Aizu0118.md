## 题目描述-Description
在H * W的矩形果园里有苹果、梨、蜜柑三种果树， 相邻（上下左右）的同种果树属于同一个区域，给出果园的果树分布，求总共有多少个区域。 （原题的样图中苹果为リ，梨为カ，蜜柑为ミ， 图中共10个区域）

### 输入-Input
多组数据，每组数据第一行为两个整数H、W（H <= 100, W <= 100）， H =0 且 W = 0代表输入结束
```
10 10
####*****@
@#@@@@#*#*
@##***@@@*
#****#*@**
##@*#@@*##
*@@@@*@@@#
***#@*@##*
*@@@*@@##@
*@*#*@##**
@****#@@#@
0 0
```
### 输出-Output
对于每组数据，输出其区域的个数。
```
33
```
### 提示-Hint
无

#include <cstdio>
#include <cmath>
#include <algorithm>
#include <queue>
#include <iostream>
#include <vector>
#include <cstdlib>
#define eps 1e-6
#define ll long long
#define INF 0x3f3f3f3f
using namespace std;

int col,row;
int cnt = 0;
char map[100][100];
int mark[100][100];
void dfs(int x,int y,char type){
    mark[x][y] = 1;
    int nextX[] = {1,0,0,-1};
    int nextY[] = {0,-1,1,0};
    for(int i = 0;i<4;i++){
        const int newX = x + nextX[i];
        const int newY = y + nextY[i];
        if(newX < 0 || newY < 0 || newX >= row || newY >= col){
            continue;
        }
        if(map[newX][newY] == type&&mark[newX][newY] == 0){
            dfs(newX,newY,type);
        }
            
    }
    
}
int main(){
    while((cin>>row>>col)){
        if(!col&&!row)
            break;
        //input
        cnt = 0;
        for(int i = 0;i<row;i++){
            for(int j = 0;j<col;j++){
                cin>>map[i][j];
                mark[i][j] = 0;
            }
        }
        for(int i = 0;i<row;i++){
            for(int j = 0;j<col;j++){
                if(mark[i][j] == 0){
                    dfs(i,j,map[i][j]);
                    cnt++;
                }
            }
        }
        cout<<cnt<<endl;
    }

    return 0;
}

## 错误分析

不能使用bfs，会报`Memory Limit Exceeded`

#include <cstdio>
#include <cmath>
#include <algorithm>
#include <queue>
#include <iostream>
#include <vector>

#define INF 0x3f3f3f3f
using namespace std;

int col,row;
int cnt = 0;
char map[100][100];
typedef pair<int,int> pos;
queue<pos> q;
void bfs(char type){
    while(q.size()>0){
        const int x = q.front().first;
        const int y = q.front().second;
        q.pop();
        map[x][y] = '.';

                continue;
            }
            if(map[newX][newY] == type){
                q.push(pos(newX,newY));
            }
        }
    }
}
